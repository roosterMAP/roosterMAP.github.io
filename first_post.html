def GetWritableMesh():
	layer_svc = lx.service.Layer()
	layer_scan = lx.object.LayerScan( layer_svc.ScanAllocate( lx.symbol.f_LAYERSCAN_EDIT ) )
	if( not layer_scan.test() ): return

	writableMeshes = set()

	for n in range( 0, layer_scan.Count() ):
		mesh = lx.object.Mesh( layer_scan.MeshEdit( n ) )
		writableMeshes.add( mesh )

	return writableMeshes
